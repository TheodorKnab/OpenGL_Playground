#version 410 core
uniform int lr;

layout (triangles) in;
layout (triangle_strip, max_vertices = 3) out;
out int gl_Layer;

in vec2 texCoordsVs[];
out vec3 TexCoords;

void main(void){
for(int i = 0; i < 256; i++){
        gl_Layer = i;
		for(int i=0; i<gl_in.length(); i++)
		  {
			
			TexCoords = vec3(texCoordsVs[i], 0);
			gl_Position = gl_in[i].gl_Position;
			EmitVertex();
		  }
        EndPrimitive();

    }
}
