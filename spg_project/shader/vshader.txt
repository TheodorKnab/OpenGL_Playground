#version 330 core
layout (location = 0) in vec3 aPos;
//layout (location = 1) in vec3 aColor;

out VS_OUT {
    vec3 color;
} vs_out;


uniform mat4 projection;
uniform mat4 view;
uniform mat4 model;

void main()
{
    vs_out.color = vec3(1,1,1);
    gl_Position = projection * view * model * vec4(aPos.x, aPos.y, aPos.z, 1.0); 
}
